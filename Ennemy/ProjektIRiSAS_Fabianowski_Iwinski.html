f <- function(x, plec = "m", przybliznormalnym = F) {
    if (is.numeric(x)) {
        if (plec == "o") {
            d = density(x, na.rm = T)
            y = x
        } else {
            l = diagnozaOsoby2011$PLEC == "mezczyzna"
            if (plec == "m") {
                y = x[l]
                d = density(y, na.rm = T)
            } else {
                l = !l
                y = x[l]
                d = density(y, na.rm = T)
            }
        }
        plot(d, xlab = "x", ylab = "gƒôsto≈õƒá(x)", main = "gƒôsto≈õƒá z pr√≥bki wybranej cechy")
        if (przybliznormalnym == T) {
            odchylenie = (var(y, na.rm = T))^(1/2)
            oczekiwana = mean(y, na.rm = T)
            cat(paste("odchylenie standardowe = ", odchylenie))
            cat(paste("\n≈õrednia = ", oczekiwana))
            dol = oczekiwana - 3 * odchylenie
            gora = oczekiwana + 3 * odchylenie
            k = seq(from = dol, to = gora, by = 0.1)
            lines(k, dnorm(k, mean = oczekiwana, sd = odchylenie), col = "red", 
                lty = 2)
            legend(120, 0.02, c("gƒôsto≈õƒá z pr√≥bki", "rozk≈Çad normalny"), col = 1:2, 
                lty = 1:2)
        }
    } else {
        cat("zly argument")
    }
}
